אִגֶּרֶת פּוֹלוֹס הַשָּׁלִיחַ אֶל־הָאֶפְסִיִּים

א   פּוֹלוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ בִּרְצוֹן אֱלֹהִים אֶל־הַקְּדשִׁים הַנִּמְצְאִים (בְּאֶפְסוֹס) וּמַאֲמִינִים בַּמָּשִׁיחַ יֵשׁוּעַ׃ 2 חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלֹהִים אָבִינוּ וַאֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃ 3 בָּרוּךְ הָאֱלֹהִים וַאֲבִי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר בֵּרְכָנוּ בְּכָל־בִּרְכַּת רוּחַ בַּמְּרוֹמִים בַּמָּשִׁיחַ׃ 4 כַּאֲשֶׁר בָּחַר אֹתָנוּ בוֹ לִפְנֵי מוֹסְדוֹת תֵּבֵל לִהְיוֹת קְדשִׁים וּתְמִימִים לְפָנָיו בְּאַהֲבָה׃ 5 יְעָדָנוּ לוֹ לְבָנִים עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ כִּרְצוֹן חֶפְצוֹ׃ 6 לִתְהִלַּת כְּבוֹד חַסְדּוֹ אֲשֶׁר־חָנַן אֹתָנוּ בִּידִידוֹ׃ 7 אֲשֶׁר־בּוֹ לָנוּ הַפִּדְיוֹם בְּדָמוֹ וּסְלִיחַת הַפְּשָׁעִים כְּרֹב חַסְדּוֹ׃ 8 אֲשֶׁר בָּא עָלֵינוּ בְּשֶׁפַע־חָכְמָה וְהַשְׂכֵּל׃ 9 וְהוֹדִיעַ אֹתָנוּ אֶת־סוֹד חֶפְצוֹ כִּרְצוֹנוֹ וַעֲצָתוֹ מֵרֹאשׁ אֲשֶׁר יָעַץ בּוֹ׃ 10 עַל־דְּבַר הַנְהָגָתוֹ בִּמְלֹאת הַיָּמִים לְקַבֵּל אֶת־הַכֹּל בַּמָּשִׁיחַ מַה־שֶּׁבַּשָּׁמַיִם וּמַה־שֶּׁבָּאָרֶץ׃ 11 וּבוֹ לָקַחְנוּ נַחֲלָתֵנוּ אֲנַחְנוּ הַמְיֻעָדִים לָהּ מִלְּפָנִים בְּמַחֲשֶׁבֶת פֹּעֵל הַכֹּל כַּעֲצַת חֶפְצוֹ׃ 12 לִהְיוֹת לִתְהִלַּת כְּבוֹדוֹ אֲנַחְנוּ אֲשֶׁר יִחַלְנוּ אֶל־הַמָּשִׁיחַ מֵאָז׃ 13 וַאֲשֶׁר גַּם־אַתֶּם שְׁתוּלִים בּוֹ אַחֲרֵי שָׁמְעֲכֶם דְּבַר הָאֱמֶת אֶת־בְּשׂוֹרַת יְשׁוּעַתְכֶם וַאֲשֶׁר בּוֹ כְּשֶׁהֶאֱמַנְתֶּם גַּם־נֶחְתַּמְתֶּם בְּרוּחַ הַהַבְטָחָה רוּחַ הַקֹּדֶשׁ׃ 14 כִּי זֶה עֶרְבוֹן נַחֲלָתֵנוּ לִפְדּוֹת־לוֹ עַם סְגֻלָּה לִתְהִלַּת כְּבוֹדוֹ׃ 15 בַּעֲבוּר זֹאת גַּם־אָנֹכִי אַחֲרֵי שָׁמְעִי אֱמוּנַתְכֶם בַּאֲדֹנֵינוּ יֵשׁוּעַ וְהָאַהֲבָה אֲשֶׁר־אֲהַבְתֶּם אֶת־כָּל־הַקְּדשִׁים׃ 16 לֹא אֶחְדַּל מֵהוֹדוֹת בַּעֲבוּרְכֶם בְּהַזְכִּירִי אֶתְכֶם בִּתְפִלּתַי׃ 17 שֶׁיִּתֵּן לָכֶם אֱלֹהֵי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲבִי הַכָּבוֹד אֶת־רוּחַ הַחָכְמָה וְהֶחָזוֹן לָדַעַת אֹתוֹ׃ 18 וְיָאֵר עֵינֵי לְבַבְכֶם לְמַעַן תֵּדְעוּ מַה־הִיא תּוֹחֶלֶת קְרִיאָתוֹ וּמָה־עשֶׁר כְּבוֹד נַחֲלָתוֹ בַּקְּדשִׁים׃ 19 וּמַה־יִּתְרוֹן גְּדֻלַּת גְּבוּרָתוֹ בָּנוּ הַמַּאֲמִינִים כְּפִי פְעֻלַּת עֹצֶם כֹּחוֹ׃ 20 אֲשֶׁר פָּעַל בַּמָּשִׁיחַ בְּהָעִיר אֹתוֹ מִן־הַמֵּתִים וַיּוֹשִׁיבֶנּוּ לִימִינוֹ בַּמְּרוֹמִים׃ 21 מִמַּעַל לְכָל־שְׂרָרָה וְשָׁלְטָן וּגְבוּרָה וּמֶמְשָׁלָה וְכֹל הַנִּקְרָא בְשֵׁם לֹא־בָעוֹלָם הַזֶּה בִּלְבַד כִּי אִם־גַּם בָּעוֹלָם הַבָּא׃ 22 וַיָּשֶׁת כֹּל תַּחַת רַגְלָיו וַיִּתֵּן אֹתוֹ לְרֹאשׁ לְכֹל אֶל־הָעֵדָה׃ 23 אֲשֶׁר הִיא גּוּפוֹ מְלוֹא שֶׁל־הַמְמַלֵּא אֶת־הַכֹּל בַּכֹּל׃

ב   גַּם־אֶתְכֶם אֲשֶׁר מֵתִים הֱיִיתֶם בְּפִשְׁעֵיכֶם וְחַטֹּאתֵיכֶם׃ 2 אֲשֶׁר הִתְהַלַּכְתֶּם בָּהֶם כְּדוֹר הָעוֹלָם הַזֶּה כִּרְצוֹן שַׂר מֶמְשֶׁלֶת הָאֲוִיר וְהוּא הָרוּחַ הַפֹּעֵל כָּעֵת בִּבְנֵי הַמֶּרִי׃ 3 וְגַם־אֲנַחְנוּ כֻלָּנוּ בְּתוֹכָם הָלַכְנוּ לְפָנִים בְּתַאֲוֹת הַבָּשָׂר לַעֲשׂוֹת חֶפְצֵי בְשָׂרֵנוּ וּמַחְשְׁבוֹתֵינוּ וַנְּהִי בְנֵי־רֹגֶז בְּטִבְעֵנוּ כִּשְׁאָר בְּנֵי אָדָם׃ 4 אֲבָל־הָאֱלֹהִים הַמָּלֵא רַחֲמִים בְּרֹב אַהֲבָתוֹ אֲשֶׁר אָהַב אֹתָנוּ׃ 5 אַחֲרֵי הֱיוֹתֵנוּ מֵתִים בַּפְּשָׁעִים הֶחֱיָנוּ עִם־הַמָּשִׁיחַ בַּחֶסֶד נוֹשַׁעְתֶּם׃ 6 וַיְעִירֵנוּ אִתּוֹ אַף־הוֹשִׁיבָנוּ בַּמְּרוֹמִים בַּמָּשִׁיחַ יֵשׁוּעַ׃ 7 לְהַרְאוֹת בַּדֹּרוֹת הַבָּאִים אֶת־גְּדֻלַּת עשֶׁר חַסְדּוֹ בְּטוֹבָתוֹ עָלֵינוּ בַּמָּשִׁיחַ יֵשׁוּעַ׃ 8 כִּי־בַחֶסֶד נוֹשַׁעְתֶּם עַל־יְדֵי הָאֱמוּנָה וְלֹא מִיֶּדְכֶם הָיְתָה זֹּאת כִּי־מַתַּת אֱלֹהִים הִיא׃ 9 לֹא מִתּוֹךְ הַמַּעֲשִׂים שֶׁלּא יִתְהַלֵּל אִישׁ׃ 10 כִּי־פֹעַל אֱלֹהִים אֲנַחְנוּ נִבְרָאִים בַּמָּשִׁיחַ יֵשׁוּעַ לְמַעֲשִׂים טוֹבִים אֲשֶׁר הֵכִין הָאֱלֹהִים מִקֶּדֶם לְמַעַן נִתְהַלֵּךְ בָּהֶם׃ 11 עַל־כֵּן זִכְרוּ כִּי־אַתֶּם מִתְּחִלָּה הַגּוֹיִם בַּבָּשָׂר הַנִּקְרָאִים עֲרֵלִים בְּפִי הַנִּקְרָאִים בְּנֵי הַמִּילָה שֶׁהִיא מַעֲשֵׂה יָדַיִם בַּבָּשָׂר׃ 12 כִּי־אַתֶּם בָּעֵת הַהִיא הֱיִיתֶם בְּלִי מָשִׁיחַ זָרִים מֵעֲדַת יִשְׂרָאֵל וְנָכְרִים לִבְרִיתוֹת הַהַבְטָחָה בְּאֵין תִּקְוָה וּבְאֵין לָכֶם אֱלֹהִים בָּעוֹלָם׃ 13 וְעַתָּה בְּיֵשׁוּעַ הַמָּשִׁיחַ אַתֶּם הָרְחוֹקִים מֵאָז הֱיִיתֶם קְרוֹבִים בְּדַם הַמָּשִׁיחַ׃ 14 כִּי הוּא שְׁלוֹמֵנוּ אֲשֶׁר עָשָׂה הַשְּׁנַיִם אֶחָד וְהָרַס מְחִיצַת הַגָּדֵר׃ 15 בְּבַטְּלוֹ הָאֵיבָה בִּבְשָׂרוֹ אֶת־תּוֹרַת הַמִּצְוֹת וְהַחֻקּוֹת לִבְרֹא בְנַפְשׁוֹ אֶת־הַשְּׁנַיִם לְאָדָם אֶחָד חָדָשׁ וַיַּעַשׂ שָׁלוֹם׃ 16 וַיְרַצֶּה אֶת־שְׁנֵיהֶם בְּגוּף אֶחָד לֵאלֹהִים עַל־יְדֵי צְלִיבָתוֹ בַּהֲמִיתוֹ בְנַפְשׁוֹ אֶת־הָאֵיבָה׃ 17 וַיָּבֹא וַיְבַשֵׂר לָכֶם שָׁלוֹם שָׁלוֹם לָרָחוֹק וְלַקָּרוֹב׃ 18 כִּי עַל־יָדוֹ יֵשׁ לִשְׁנֵינוּ מָבוֹא בְּרוּחַ אֶחָד אֶל־אָבִינוּ׃ 19 לָכֵן אֵינְכֶם עוֹד גֵּרִים וְתוֹשָׁבִים כִּי־בְנֵי־עִירָם שֶׁל־הַקְּדשִׁים וּבְנֵי בֵּית אֱלֹהִים אַתֶּם׃ 20 בְּנוּיִם עַל־יְסוֹד הַשְּׁלִיחִים וְהַנְּבִיאִים וְיֵשׁוּעַ הַמָּשִׁיחַ הוּא אֶבֶן הַפִּנָּה׃ 21 אֲשֶׁר חֻבַּר־בּוֹ יַחַד הַבִּנְיָן כֻּלּוֹ עַד אֲשֶׁר־יִגְבַּהּ לְהֵיכַל קֹדֶשׁ לַיהוָֹה׃ 22 וּבוֹ נִבְנִים גַּם־אַתֶּם לִהְיוֹת מִשְׁכַּן אֱלֹהִים בָּרוּחַ׃

ג   בַּעֲבוּר זֹאת אֲנִי פוֹלוֹס אֲסִיר הַמָּשִׁיחַ לְמַעַנְכֶם הַגּוֹיִם׃ 2 כִּי־אָמְנָם שְׁמַעְתֶּם הַנְהָגַת חֶסֶד הָאֱלֹהִים אֲשֶׁר־נִתַּן לִי אֲלֵיכֶם׃ 3 כִּי בֶחָזוֹן גִּלָּה־לִי הַסּוֹד אֲשֶׁר כָּתַבְתִּי לָכֶם בִּמְעַט מִלִּין׃ 4 וּבְקָרְאֲכֶם תּוּכְלוּ לְהַכִּיר אֶת־בִּינָתִי בְּסוֹד הַמָּשִׁיחַ׃ 5 אֲשֶׁר בַּדּוֹרֹת הָרִאשֹׁנִים לֹא נוֹדַע לִבְנֵי אָדָם כַּאֲשֶׁר נִגְלָה עַתָּה לִשְׁלִיחָיו הַקְּדשִׁים וְלִנְבִיאָיו בָּרוּחַ׃ 6 שֶׁיִּהְיוּ הַגּוֹיִם גַּם־הֵם נֹחֲלִים וּבְגוּף אֶחָד וְחַבְרֵי הַהַבְטָחָה בַּמָּשִׁיחַ יֵשׁוּעַ עַל־יְדֵי הַבְּשׂרָה׃ 7 אֲשֶׁר הָיִיתִי לָהּ לִמְשָׁרֵת בְּמַתְּנַת חֶסֶד אֱלֹהִים הַנְּתוּנָה לִי כְּפִי פְּעֻלַּת גְּבוּרָתוֹ׃ 8 לִי אֲנִי צְעִיר הַצְּעִירִים שֶׁבְּכָל־הַקְּדשִׁים נִתַּן הַחֶסֶד הַזֶּה לְבַשֵׂר בַּגּוֹיִם אֶת־עשֶׁר הַמָּשִׁיחַ אֲשֶׁר רַב הוּא לְאֵין חֵקֶר׃ 9 וּלְהָאִיר עֵינֵי כֹל לִרְאוֹת מַה־הַנְהָגַת הַסּוֹד הַנִּסְתָּר מִדֹּרֹת עוֹלָם בֵּאלֹהִים יוֹצֵר הַכֹּל (עַל־יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ)׃ 10 לְמַעַן
